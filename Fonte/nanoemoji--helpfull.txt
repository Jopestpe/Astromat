absl.app:
  -?,--[no]help: show this help
    (default: 'false')
  --[no]helpfull: show full help
    (default: 'false')
  --[no]helpshort: show this help
    (default: 'false')
  --[no]helpxml: like --helpfull, but generates XML output
    (default: 'false')
  --[no]only_check_args: Set to true to validate args and exit.
    (default: 'false')
  --[no]pdb: Alias for --pdb_post_mortem.
    (default: 'false')
  --[no]pdb_post_mortem: Set to true to handle uncaught exceptions with PDB post mortem.
    (default: 'false')
  --profile_file: Dump profile information to a file (for python -m pstats). Implies --run_with_profiling.
  --[no]run_with_pdb: Set to true for PDB debug mode
    (default: 'false')
  --[no]run_with_profiling: Set to true for profiling the script. Execution will be slower, and the output format might change over
    time.
    (default: 'false')
  --[no]use_cprofile_for_profiling: Use cProfile instead of the profile module for profiling. This has no effect unless
    --run_with_profiling is set.
    (default: 'true')

absl.logging:
  --[no]alsologtostderr: also log to stderr?
    (default: 'false')
  --log_dir: directory to write logfiles into
    (default: '')
  --logger_levels: Specify log level of loggers. The format is a CSV list of `name:level`. Where `name` is the logger name used
    with `logging.getLogger()`, and `level` is a level name  (INFO, DEBUG, etc). e.g. `myapp.foo:INFO,other.logger:DEBUG`
    (default: '')
  --[no]logtostderr: Should only log to stderr?
    (default: 'false')
  --[no]showprefixforinfo: If False, do not prepend prefix to info messages when it's logged to stderr, --verbosity is set to INFO
    level, and python logging is used.
    (default: 'true')
  --stderrthreshold: log messages at this level, or more severe, to stderr in addition to the logfile.  Possible values are
    'debug', 'info', 'warning', 'error', and 'fatal'.  Obsoletes --alsologtostderr. Using --alsologtostderr cancels the effect of
    this flag. Please also note that this flag is subject to --verbosity and requires logfile not be stderr.
    (default: 'fatal')
  -v,--verbosity: Logging verbosity level. Messages logged at this level or lower will be included. Set to 1 for debug logging. If
    the flag was not set or supplied, the value will be changed from the default of -1 (warning) to 0 (info) after flags are
    parsed.
    (default: '-1')
    (an integer)

nanoemoji.config:
  --ascender: Ascender
    (an integer)
  --bitmap_resolution: Resolution of bitmap in pixels. Always square for now.
    (an integer)
  --[no]clip_to_viewbox: Whether to clip content outside viewbox.
  --clipbox_quantization: Whether to quantize COLR clip boxes to multiples of positive integer, i.e. rounding {x,y}Min => -Inf
    (floor) and {x,y}Max => +Inf (ceiling). By default, it's 2% of UPEM (e.g. multiples of 20 units out of 1024).
    (a positive integer)
  --color_format: <cbdt|cff2_colr_0|cff2_colr_1|cff_colr_0|cff_colr_1|glyf|glyf_colr_0|glyf_colr_1|picosvg|picosvgz|sbix|untoucheds
    vg|untouchedsvgz>: Type of font to generate.
  --descender: Descender.
    (an integer)
  --family: Family name.
  --fea_file: Feature file.
  --glyphmap_generator: A program that takes a list of filenames and outputs a file csv whose rows contain filename, codepoint(s),
    glyph name.
  --[no]ignore_reuse_error: Whether to fail or continue with a warning when picosvg cannot compute affine between paths that
    normalize the same.
  --[no]keep_glyph_names: Whether or not to store glyph names in the font.
  --linegap: Line gap.
    (an integer)
  --output_file: Output filename.
  --pngquant_flags: Additional options to pass on to pngquant.
  --[no]pretty_print: Whether to prefer pretty printed content whenever possible (for testing).
  --reuse_tolerance: Allowable absolute difference in reused shape in input coordinates (e.g. svg). Normalized shapes snap to whole
    multiples of tolerance; A negative value means that shape reuse is disabled.
    (a number)
  --transform: User transform, in font coordinates.
  --upem: Units per em.
    (an integer)
  --[no]use_pngquant: Whether or not to quantize PNGs using pngquant.
  --[no]use_zopflipng: Whether or not to compress PNGs using zopfli.
  --version_major: Major version.
    (an integer)
  --version_minor: Minor version.
    (an integer)
  --width: Width.
    (an integer)
